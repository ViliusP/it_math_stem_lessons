\begin{tikzpicture}[baseline=0]
    % Draw the circle with radius 3
    \draw[line width=.75pt] (0,0) circle (3cm);
    
    % Define points
    \coordinate (O) at (0,0);
    \coordinate (A) at (3,0);
    \coordinate (B) at ({3*cos(60)}, {3*sin(60)});
    
    % Draw radii
    \draw[line width=1.05pt] (O) -- (A);
    \draw[line width=1.05pt] (O) -- (B);
    
    % Draw chord
    \draw[line width=1.05pt] (A) -- (B);
    
    % Draw arc
    \draw[line width=3pt] (A) arc[start angle=0,end angle=60,radius=3cm];
    
    % Mark the center of the circle with a dot
    \fill (O) circle (2pt);
    \node at (O) [below left] {$O$};
    
    % Draw angle at the circumference
    \pic [draw, "$x$", angle eccentricity=1.3, angle radius=.65cm] {angle = B--A--O};
\end{tikzpicture}
